```yml
# 1. workflow 이름  
name: workflow_sample  
  
# 이벤트 트리거  
on:  
  push:  
    branches: [ "main" ]  
    # main branch에 push 되었을 때  
  pull_request:  
    branches: [ "main" ]  
    # main branch로의 PullRequest가 생성되거나 업데이트 될 경우  
  schedule:  
    - cron: '0 0 * * *'  
    # 정기적인 trigger  workflow_run:  
    workflows: ['another_workflow']  
    types:  
      - completed  
      - in_progress  
    # 다른 workflow의 성공/실패에 따라 실행될 때  
  workflow_call:  
    # 다른 workflow에서 해당 workflow를 호출 할 수 있는 flag  workflow_dispatch:  
    inputs:  
      environment:  
        description:  
        required: true  
        default: 'prod'  
        type: choice  
        options:  
          - prod  
          - stage  
          - dev  
  # 수동으로 workflow를 실행 - 하위에 inputs가 추가됨  
  
jobs:  
  build:  
    runs-on: ubuntu-latest  
    permissions:  
      contents: read  
  
    steps:  
      - uses: actions/checkout@v4  
      - name: Set up JDK 17  
        uses: actions/setup-java@v4  
        with:  
          java-version: '17'  
          distribution: 'temurin'  
  
      - name: Setup Gradle  
        uses: gradle/actions/setup-gradle@af1da67850ed9a4cedd57bfd976089dd991e2582  
  
      - name: Build with Gradle Wrapper  
        run: ./gradlew build  
  
  dependency-submission:  
    runs-on: ubuntu-latest  
    permissions:  
      contents: write  
  
    steps:  
      - uses: actions/checkout@v4  
      - name: Set up JDK 17  
        uses: actions/setup-java@v4  
        with:  
          java-version: '17'  
          distribution: 'temurin'  
  
      - name: Generate and submit dependency graph  
        uses: gradle/actions/dependency-submission@af1da67850ed9a4cedd57bfd976089dd991e2582 # v4.0.0
```
